# A Romaji
  
```cpp
#include<iostream>
  
using namespace std;
  
string str;
  
bool check(char c)
{
    if(c=='a' || c=='o' || c=='u' || c=='i' || c=='e')
        return true;
    else
        return false;
}
  
int main()
{
    cin>>str;
    for(int i=0;i<str.size();i++)
    {
        if(str[i]=='n')
            continue;
        else if(check(str[i]))
            continue;
        else if(!check(str[i+1]))
        {
            cout<<"NO"<<endl;
            return 0;
        }
    }
    cout<<"YES"<<endl;
    return 0;
}
```

# B Turn the Rectangles
  
```cpp
#include<iostream>
#include<algorithm>
  
using namespace std;
  
struct node
{
    int x,y;
}a[100003];
  
int n;
  
int main()
{
    ios::sync_with_stdio(false);
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i].x>>a[i].y;
    int h=max(a[1].x,a[1].y);
    for(int i=1;i<n;i++)
    {
        if(h>=a[i+1].x && h>=a[i+1].y)
        {
            h=max(a[i+1].x,a[i+1].y);
            continue;
        }
        else if(h>=a[i+1].x)
        {
            h=a[i+1].x;
            continue;
        }
        else if(h>=a[i+1].y)
        {
            h=a[i+1].y;
            continue;
        }
        else
        {
            cout<<"NO"<<endl;
            return 0;
        }
    }
    cout<<"YES"<<endl;
    return 0;
}
```

# C Reorder the Array
  
```cpp
#include<iostream>
#include<algorithm>
  
using namespace std;
  
int n,ans,tmp;
int a[100003];
  
bool cmp(int a,int b)
{
    return a>b;
}
  
int main()
{
    ios::sync_with_stdio(false);
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    sort(a+1,a+n+1);
    for(int i=0;i<=n;i++)
    {
        if(a[i]==a[i+1])
            tmp++;
        else
        {
            ans=max(ans,tmp);
            tmp=1;
        }
    }
    cout<<n-ans<<endl;
    return 0;
}
```
